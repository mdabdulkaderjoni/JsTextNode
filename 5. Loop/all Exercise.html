        function octalToDecimal(octalnum)
        {
            let decimalnum = 0, temp = 0;
    
            while(octalnum != 0)
            {
                decimalnum = decimalnum + (octalnum%10) * Math.pow(8,temp);
                temp++;
                octalnum = octalnum / 10;
            }
    
        return decimalnum;
