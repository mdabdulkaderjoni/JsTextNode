program to read weekday number and print weekday name using switch.


    let wDay=Number(prompt('Enter a number between 0-7'))
     
    switch(wDay)
    {
        case 0: 
            console.log("Sunday");
            break;
        case 1: 
            console.log("Monday");
            break;
        case 2: 
            console.log("Tuesday");
            break;
        case 3: 
            console.log("Wednesday");
            break;
        case 4: 
            console.log("Thursday");
            break;
        case 5: 
            console.log("Friday");
            break;
        case 6:

            break;

            console.log("Invalid weekday number.");
    }

Check VOWEL or CONSONANT program using switch


    let ch=String(prompt('Enter a character'))    

    //condition to check character is alphabet or not
    if((ch>='A' && ch<='Z') || (ch>='a' && ch<='z'))
    {
        //check for VOWEL or CONSONANT
        switch(ch)
        {
            case 'A':
            case 'E':
            case 'I':
            case 'O':
            case 'U':
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                console.log(`${ch} is vowel`)
                break;
            default:
                console.log(`${ch} is consonant`)            
        }
    }
    else
    {
        console.log(`${ch} is not an alphabet`)
    }
 


Calculator program with Basic operations using switch



    let num1, num2, result, ch;
    num1=Number(prompt('Enter a number'))
    num2=Number(prompt('Enter another number'))
    ch=prompt('Choose operation to perform (+,-,*,/,%):')
     
    result=0;
    switch(ch)    
    {
        case '+':
            result=num1+num2;
            break;
             
        case '-':
            result=num1-num2;
            break;
         
        case '*':
            result=num1*num2;
            break;
             
        case '/':
            result=num1/num2;
            break;
             
        case '%':
            result=num1%num2;
            break;
        default:
            console.log("Invalid operation.\n");
    }
 console.log(`${num1} ${ch} ${num2} is: ${result}`)

A menu-driven program to compute the area of the various geometrical shape



      let choice,r,l,w,b,h;
      let area;
    
      choice=parseInt(prompt('Input 1 for area of circle\n
                              Input 2 for area of rectangle\n
                              Input 3 for area of triangle\n'))
      switch(choice)
      {
           case 1:
                 r=Number(prompt("Input radious of the circle : "));
            
                 area=3.14*r*r;
                 break;
            case 2:
                  l=Number(prompt("Input length of the rectangle : "));
                  w=Number(prompt("Input width of the rectangle : "));
                  area=l*w;
                  break;
            case 3:
                  
                  b=Number(prompt("Input base : "));
                  h=Number(prompt("Input height : "));
                  area=.5*b*h;
                  break;
          }
          console.log(`Area is: ${area}`)
}
